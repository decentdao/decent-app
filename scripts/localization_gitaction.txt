name: Verify Localization Keys

on:
  pull_request:
    branches:
      - main

jobs:
  verify-localization:
    runs-on: ubuntu-latest

    steps:
      # Step 1: Checkout the repository
      - name: Checkout code
        uses: actions/checkout@v3

      # Step 2: Set up Node.js
      - name: Set up Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '16' # Use the appropriate Node.js version for your project

      # Step 3: Install dependencies
      - name: Install dependencies
        run: npm install

      # Step 4: Run the localization_verify.js script
      - name: Run localization verification
        run: node scripts/localization_verify.js

      # Step 5: Fail the workflow if checkKeys returns false
      - name: Check result
        run: |
          if ! node scripts/localization_verify.js | grep -q "All keys present in all language files: true"; then
            echo "Localization verification failed. Missing keys or files detected."
            exit 1
          fi
